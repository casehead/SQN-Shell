#!/usr/bin/env bash
# ============================================================
#  SQN Shell v2.0.0 — Linux Overlay
#  Sine Qua Non — Portable SD-card-resident shell ecosystem
#  Stateless on host. All tools local to SD.
#
#  Usage:
#    ./sqn                    # Enter subshell
#    source ./sqn             # Load into current shell
#    sqn bootstrap            # Initialize directory structure
#    sqn update               # Query upstream, build upgrade plan
#    sqn upgrade              # Download and install tools
#    sqn verify               # Check installed tool integrity
#    sqn run <command>        # Execute in SQN environment
#    sqn doctor               # Full diagnostics
#    sqn unload               # Restore host shell state
#    sqn clean-bin            # Clean up extracted bin clutter
#    sqn explain <command>    # Show help for a command
# ============================================================

set -uo pipefail

# ============================================================
# CONFIGURATION
# ============================================================
export SQN_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
export SQN_BIN="${SQN_ROOT}/bin"
export SQN_LIB="${SQN_ROOT}/lib"
export SQN_HELPERS="${SQN_LIB}/helpers"
export SQN_CACHE="${SQN_ROOT}/cache"
export SQN_SESSIONS="${SQN_ROOT}/sessions"
export SQN_LOG="${SQN_ROOT}/sqn.log"
export SQN_VERSION="2.0.0"
export SQN_PLATFORM="linux"
export SQN_NAME="Sine Qua Non"

# Security
export SQN_ALLOW_SUDO="${SQN_ALLOW_SUDO:-0}"
export SQN_DRY_RUN="${SQN_DRY_RUN:-0}"

# Rust toolchain PATH — inject early so detection finds it
_sqn_cargo_path="${HOME}/.cargo/bin"
[[ -d "$_sqn_cargo_path" ]] && [[ ":$PATH:" != *":$_sqn_cargo_path:"* ]] && export PATH="$_sqn_cargo_path:$PATH"
unset _sqn_cargo_path

# ============================================================
# DIRECTORY CONTRACT
# ============================================================
_sqn_ensure_dirs() {
    mkdir -p "$SQN_BIN/x86_64" "$SQN_BIN/aarch64" "$SQN_BIN/armv7" 2>/dev/null
    mkdir -p "$SQN_LIB" "$SQN_HELPERS" "$SQN_CACHE" "$SQN_SESSIONS" 2>/dev/null
    mkdir -p "$SQN_LIB/docs" 2>/dev/null
}
_sqn_ensure_dirs

# ============================================================
# LOGGING
# ============================================================
_sqn_log()     { echo "[$(date '+%Y-%m-%d %H:%M:%S')] [$1] $2" >> "$SQN_LOG" 2>/dev/null; }
_sqn_info()    { echo -e "\033[1;36m➜ $1\033[0m";    _sqn_log "INFO"  "$1"; }
_sqn_success() { echo -e "\033[1;32m✓ $1\033[0m";    _sqn_log "OK"    "$1"; }
_sqn_warn()    { echo -e "\033[1;33m⚠ $1\033[0m";    _sqn_log "WARN"  "$1"; }
_sqn_error()   { echo -e "\033[1;31m✗ $1\033[0m" >&2; _sqn_log "ERROR" "$1"; }

# --- Architecture ---
_sqn_detect_arch() {
    case "$(uname -m)" in
        x86_64)         echo "x86_64" ;;
        aarch64|arm64)  echo "aarch64" ;;
        armv7l)         echo "armv7" ;;
        i686|i386)      echo "x86" ;;
        *)              echo "unknown" ;;
    esac
}
export SQN_ARCH="$(_sqn_detect_arch)"
export SQN_ARCH_BIN="${SQN_BIN}/${SQN_ARCH}"

# --- Distro / Package Manager ---
_sqn_detect_distro() {
    if   command -v apt-get >/dev/null 2>&1; then echo "apt"
    elif command -v pacman  >/dev/null 2>&1; then echo "pacman"
    elif command -v dnf     >/dev/null 2>&1; then echo "dnf"
    elif command -v zypper  >/dev/null 2>&1; then echo "zypper"
    elif command -v apk     >/dev/null 2>&1; then echo "apk"
    elif command -v xbps-install >/dev/null 2>&1; then echo "xbps"
    else echo "unknown"
    fi
}
export SQN_DISTRO="$(_sqn_detect_distro)"

# --- Display / GUI ---
_sqn_detect_display() {
    if   [[ -n "${WAYLAND_DISPLAY:-}" ]];  then echo "wayland"
    elif [[ -n "${DISPLAY:-}" ]];          then echo "x11"
    else echo "headless"
    fi
}
export SQN_DISPLAY="$(_sqn_detect_display)"

# --- Privilege ---
export SQN_ROOT_ACCESS
[[ $EUID -eq 0 ]] && SQN_ROOT_ACCESS="yes" || SQN_ROOT_ACCESS="no"

# --- Language / Toolchain Detection ---
_sqn_detect_toolchains() {
    # Python
    if command -v python3 >/dev/null 2>&1; then
        export SQN_PYTHON="yes"
        export SQN_PYTHON_BIN="$(command -v python3)"
    elif command -v python >/dev/null 2>&1; then
        export SQN_PYTHON="yes"
        export SQN_PYTHON_BIN="$(command -v python)"
    else
        export SQN_PYTHON="no"
        export SQN_PYTHON_BIN=""
    fi

    # C compiler
    if command -v gcc >/dev/null 2>&1; then
        export SQN_C="yes"
        export SQN_C_BIN="$(command -v gcc)"
    elif command -v clang >/dev/null 2>&1; then
        export SQN_C="yes"
        export SQN_C_BIN="$(command -v clang)"
    else
        export SQN_C="no"
        export SQN_C_BIN=""
    fi

    if command -v rustc >/dev/null 2>&1; then
        export SQN_RUST="yes"
        export SQN_RUST_BIN="$(command -v rustc)"
        export SQN_CARGO_BIN="$(command -v cargo 2>/dev/null || echo '')"
    else
        export SQN_RUST="no"
        export SQN_RUST_BIN=""
        export SQN_CARGO_BIN=""
    fi

    # Node
    if command -v node >/dev/null 2>&1; then
        export SQN_NODE="yes"
        export SQN_NODE_BIN="$(command -v node)"
    else
        export SQN_NODE="no"
        export SQN_NODE_BIN=""
    fi
}
_sqn_detect_toolchains

# ============================================================
# WRITABILITY GUARD
# ============================================================
_sqn_require_writable() {
    if [[ ! -w "$SQN_ROOT" ]]; then
        _sqn_error "SD card is read-only. Cannot write to $SQN_ROOT"
        return 1
    fi
    return 0
}

# ============================================================
# SMART WRAPPERS — prefer card tools over system tools
# ============================================================
_sqn_ls_smart() {
    if [[ -x "${SQN_ARCH_BIN}/eza" ]]; then
        "${SQN_ARCH_BIN}/eza" -la --group-directories-first --git "$@" 2>/dev/null || "${SQN_ARCH_BIN}/eza" "$@"
    else
        ls -la --color=auto "$@" 2>/dev/null || ls -la "$@"
    fi
}

_sqn_grep_smart() {
    if [[ -x "${SQN_ARCH_BIN}/rg" ]]; then "${SQN_ARCH_BIN}/rg" "$@"
    else grep --color=auto "$@"
    fi
}

_sqn_cat_smart() {
    if [[ -x "${SQN_ARCH_BIN}/bat" ]]; then "${SQN_ARCH_BIN}/bat" --paging=never "$@"
    else cat "$@"
    fi
}

_sqn_find_smart() {
    if [[ -x "${SQN_ARCH_BIN}/fd" ]]; then "${SQN_ARCH_BIN}/fd" "$@"
    else find "$@"
    fi
}

_sqn_find_normalized() {
    local pattern="${1:?_sqn_find_normalized requires a pattern}"
    local path="${2:-.}"
    if [[ -x "${SQN_ARCH_BIN}/fd" ]]; then
        "${SQN_ARCH_BIN}/fd" "$pattern" "$path"
    else
        find "$path" -name "*${pattern}*" 2>/dev/null
    fi
}

_sqn_ps_smart() {
    if [[ -x "${SQN_ARCH_BIN}/procs" ]]; then "${SQN_ARCH_BIN}/procs" "$@"
    else ps aux "$@"
    fi
}

_sqn_df_smart() {
    if [[ -x "${SQN_ARCH_BIN}/dust" ]]; then "${SQN_ARCH_BIN}/dust" "$@"
    else df -h "$@"
    fi
}

_sqn_du_smart() {
    local target="${1:-.}"
    if [[ -x "${SQN_ARCH_BIN}/dust" ]]; then "${SQN_ARCH_BIN}/dust" -r "$target" 2>/dev/null
    else du -sh "$target"/* 2>/dev/null | sort -rh
    fi
}

# ============================================================
# PACKAGE MANAGER WRAPPERS — distro-agnostic
# ============================================================
_sqn_pkg_install() {
    case "$SQN_DISTRO" in
        apt)    sudo apt-get install -y "$@" ;;
        pacman) sudo pacman -S --noconfirm "$@" ;;
        dnf)    sudo dnf install -y "$@" ;;
        zypper) sudo zypper install -y "$@" ;;
        apk)    sudo apk add "$@" ;;
        xbps)   sudo xbps-install -y "$@" ;;
        *)      _sqn_error "Unknown package manager. Install manually: $*"; return 1 ;;
    esac
}

_sqn_pkg_remove() {
    case "$SQN_DISTRO" in
        apt)    sudo apt-get remove -y "$@" ;;
        pacman) sudo pacman -R --noconfirm "$@" ;;
        dnf)    sudo dnf remove -y "$@" ;;
        zypper) sudo zypper remove -y "$@" ;;
        apk)    sudo apk del "$@" ;;
        xbps)   sudo xbps-remove -y "$@" ;;
        *)      _sqn_error "Unknown package manager. Remove manually: $*"; return 1 ;;
    esac
}

_sqn_pkg_update() {
    case "$SQN_DISTRO" in
        apt)    sudo apt-get update && sudo apt-get upgrade -y ;;
        pacman) sudo pacman -Syu --noconfirm ;;
        dnf)    sudo dnf upgrade -y ;;
        zypper) sudo zypper update -y ;;
        apk)    sudo apk update && sudo apk upgrade ;;
        xbps)   sudo xbps-install -Su ;;
        *)      _sqn_error "Unknown package manager."; return 1 ;;
    esac
}

_sqn_pkg_search() {
    case "$SQN_DISTRO" in
        apt)    apt-cache search "$@" ;;
        pacman) pacman -Ss "$@" ;;
        dnf)    dnf search "$@" ;;
        zypper) zypper search "$@" ;;
        apk)    apk search "$@" ;;
        xbps)   xbps-query -Rs "$@" ;;
        *)      _sqn_error "Unknown package manager."; return 1 ;;
    esac
}

    ["eza"]="1"

_sqn_get_release() {
    local repo="$1"
    local token="${GITHUB_TOKEN:-}"
    local auth_header=""
    [[ -n "$token" ]] && auth_header="Authorization: token $token"

    curl -sL --connect-timeout 10 --retry 2 \
        ${auth_header:+-H "$auth_header"} \
        "https://api.github.com/repos/$repo/releases/latest" 2>/dev/null
}

_sqn_extract_url() {
    local json="$1" pattern="$2"
    echo "$json" | grep -oP '"browser_download_url":\s*"\K[^"]+' | grep -E "$pattern" | head -1
}

_sqn_extract_tag() {
    local json="$1"
    echo "$json" | grep -oP '"tag_name":\s*"\K[^"]+' | head -1
}

# ============================================================
# BOOTSTRAP
# ============================================================
_sqn_bootstrap() {
    _sqn_info "Bootstrapping ${SQN_NAME} v${SQN_VERSION}"

    _sqn_ensure_dirs

    local env_file="$SQN_ROOT/.env"
    if [[ ! -f "$env_file" ]]; then
        cat > "$env_file" << EOF
SQN_ARCH=$SQN_ARCH
SQN_DISTRO=$SQN_DISTRO
SQN_VERSION=$SQN_VERSION
SQN_PLATFORM=$SQN_PLATFORM
SQN_DISPLAY=$SQN_DISPLAY
BOOTSTRAP_DATE=$(date -Iseconds)
EOF
        _sqn_success "Environment detected and saved"
    else
        _sqn_warn "Environment file exists, skipping"
    fi

    _sqn_init_manifest
    _sqn_init_language
    _sqn_init_helpers

    _sqn_success "Bootstrap complete. Run 'sqn update' to check for tools"
}

# ============================================================
# UPDATE (query upstream, build plan)
# ============================================================
_sqn_plan_update() {
    _sqn_require_writable || return 1

    local plan_file="$SQN_CACHE/upgrade_plan.tsv"
    mkdir -p "$SQN_CACHE" 2>/dev/null

    _sqn_info "Querying upstream for latest releases..."
    printf "# Upgrade plan generated %s\n" "$(date)" > "$plan_file"
    printf "TOOL\tSTATUS\tVERSION\tURL\n" >> "$plan_file"

    local plan_has_changes=0

    for tool in "${!SQN_UPSTREAM_REPO[@]}"; do
        echo -n "  $tool... "

        local repo="${SQN_UPSTREAM_REPO[$tool]}"
        local pattern=""

        case "$tool" in
            eza)    pattern="eza_.*_linux_${SQN_ARCH}.*\.tar\.gz" ;;
            bat)    pattern="bat-v.*-${SQN_ARCH}-unknown-linux-musl\.tar\.gz" ;;
            rg)     pattern="ripgrep-.*-${SQN_ARCH}-unknown-linux-musl\.tar\.gz" ;;
            fd)     pattern="fd-v.*-${SQN_ARCH}-unknown-linux-musl\.tar\.gz" ;;
            btop)   pattern="btop-.*-${SQN_ARCH}-linux-musl\.tbz" ;;
            dust)   pattern="dust-v.*-${SQN_ARCH}-unknown-linux-musl\.tar\.gz" ;;
            zoxide) pattern="zoxide-.*-${SQN_ARCH}-unknown-linux-musl\.tar\.gz" ;;
            fzf)
                local fzf_arch
                case "$SQN_ARCH" in
                    x86_64) fzf_arch="amd64" ;;
                    aarch64) fzf_arch="arm64" ;;
                    armv7) fzf_arch="armv7" ;;
                    x86) fzf_arch="386" ;;
                esac
                pattern="fzf-.*-linux_${fzf_arch}\.tar\.gz"
                ;;
            jq)     pattern="jq-linux-${SQN_ARCH}" ;;
            procs)  pattern="procs-v.*-${SQN_ARCH}-linux\.zip" ;;
            tldr)   pattern="tldr-linux-${SQN_ARCH}-musl" ;;
        esac

        local release_json
        release_json="$(_sqn_get_release "$repo")"

        if [[ -z "$release_json" ]] || [[ "$release_json" == *"Not Found"* ]]; then
            echo "FAIL (API error)"
            printf "%s\tERROR\tapi_failed\t\n" "$tool" >> "$plan_file"
            continue
        fi

        if [[ "$release_json" == *"API rate limit"* ]]; then
            echo "FAIL (rate limit)"
            _sqn_warn "GitHub API rate limit hit. Set GITHUB_TOKEN or wait."
            printf "%s\tERROR\trate_limit\t\n" "$tool" >> "$plan_file"
            continue
        fi

        local version url
        version="$(_sqn_extract_tag "$release_json")"
        url="$(_sqn_extract_url "$release_json" "$pattern")"

        if [[ -z "$url" ]]; then
            echo "FAIL (no matching asset)"
            printf "%s\tERROR\tno_asset\t\n" "$tool" >> "$plan_file"
            continue
        fi

        local local_path="${SQN_ARCH_BIN}/$tool"
        local status="new"

        if [[ -f "$local_path" ]]; then
            status="update_available"
            plan_has_changes=1
            echo "UPDATE ($version)"
        else
            status="missing"
            plan_has_changes=1
            echo "NEW ($version)"
        fi

        printf "%s\t%s\t%s\t%s\n" "$tool" "$status" "$version" "$url" >> "$plan_file"
    done

    if [[ $plan_has_changes -eq 0 ]]; then
        _sqn_success "All tools up to date. No changes needed."
        rm -f "$plan_file"
    else
        _sqn_success "Update plan created. Run 'sqn upgrade' to apply changes."
    fi

    return $plan_has_changes
}

# ============================================================
# UPGRADE (download and replace)
# ============================================================
_sqn_safe_write() {
    local url="$1" dest="$2"
    local tmp="${dest}.tmp.$$"
    local lock_file="$SQN_CACHE/.fetch.lock"
    local error_log="${SQN_CACHE}/extract_errors.log"

    _sqn_require_writable || return 1

    # Lockfile
    if [[ -f "$lock_file" ]]; then
        local lock_pid
        lock_pid=$(cat "$lock_file" 2>/dev/null)
        if [[ -n "$lock_pid" ]] && kill -0 "$lock_pid" 2>/dev/null; then
            _sqn_error "Another fetch operation is running (PID $lock_pid)"
            return 1
        fi
    fi
    echo $$ > "$lock_file"
    trap 'rm -f "$lock_file"' EXIT

    # Download
    if ! curl -sL --connect-timeout 15 --retry 3 "$url" -o "$tmp"; then
        _sqn_error "Failed to download: $url"
        echo "Download failed: $url" >> "$error_log"
        rm -f "$tmp"
        return 1
    fi

    local tool=$(basename "$dest")
    local depth="${SQN_STRIP_DEPTH[$tool]:-1}"
    local extract_ok=0

    # Extract based on archive type
    case "$url" in
        *.tar.gz|*.tgz)
            # Try strip-components first
            if tar -xzf "$tmp" -C "$(dirname "$dest")" --strip-components="$depth" 2>>"$error_log"; then
                extract_ok=1
            else
                # Fallback: find binary anywhere
                local extract_dir="${dest}_extract.$$"
                mkdir -p "$extract_dir"
                if tar -xzf "$tmp" -C "$extract_dir" 2>>"$error_log"; then
                    local bin_path=$(find "$extract_dir" -type f -name "$tool" 2>/dev/null | head -1)
                    if [[ -n "$bin_path" ]]; then
                        mv "$bin_path" "$dest" && extract_ok=1
                    fi
                fi
                rm -rf "$extract_dir"
            fi
            ;;
        *.tbz|*.tar.bz2)
            local extract_dir="${dest}_extract.$$"
            mkdir -p "$extract_dir"
            if tar -xjf "$tmp" -C "$extract_dir" 2>>"$error_log"; then
                local bin_path=$(find "$extract_dir" -type f -name "$tool" 2>/dev/null | head -1)
                [[ -n "$bin_path" ]] && mv "$bin_path" "$dest" && extract_ok=1
            fi
            rm -rf "$extract_dir"
            ;;
        *.zip)
            if unzip -jq "$tmp" -d "$(dirname "$dest")" 2>>"$error_log"; then
                extract_ok=1
            else
                local extract_dir="${dest}_extract.$$"
                mkdir -p "$extract_dir"
                if unzip -q "$tmp" -d "$extract_dir" 2>>"$error_log"; then
                    local bin_path=$(find "$extract_dir" -type f -name "$tool" 2>/dev/null | head -1)
                    [[ -n "$bin_path" ]] && mv "$bin_path" "$dest" && extract_ok=1
                fi
                rm -rf "$extract_dir"
            fi
            ;;
        *)
            mv "$tmp" "$dest" && extract_ok=1
            ;;
    esac

    rm -f "$tmp"

    if [[ $extract_ok -eq 0 ]]; then
        _sqn_error "Failed to extract: $(basename "$dest")"
        _sqn_error "See: $error_log"
        return 1
    fi

    chmod +x "$dest" 2>/dev/null || {
        _sqn_warn "Could not set executable permission on $dest"
    }

    _sqn_success "Installed: $(basename "$dest")"
    return 0
}

_sqn_install_binary() {
    local tool="$1" url="$2"
    _sqn_safe_write "$url" "${SQN_ARCH_BIN}/$tool"
}

_sqn_apply_upgrade() {
    local plan_file="$SQN_CACHE/upgrade_plan.tsv"

    if [[ ! -f "$plan_file" ]]; then
        _sqn_error "No upgrade plan found. Run 'sqn update' first."
        return 1
    fi

    _sqn_info "Applying upgrades from plan..."

    local upgraded=0 failed=0

    while IFS=$'\t' read -r tool status version url || [[ -n "$tool" ]]; do
        [[ "$tool" == "#"* ]] && continue
        [[ "$tool" == "TOOL" ]] && continue
        [[ -z "$tool" ]] && continue

        tool=$(echo "$tool" | xargs)
        status=$(echo "$status" | xargs)
        version=$(echo "$version" | xargs)
        url=$(echo "$url" | xargs)

        case "$status" in
            missing|update_available)
                echo -n "  $tool ($version)... "
                if _sqn_install_binary "$tool" "$url"; then
                    ((upgraded++))
                else
                    ((failed++))
                fi
                ;;
            ERROR)
                _sqn_warn "  $tool: $version"
                ((failed++))
                ;;
        esac
    done < "$plan_file"

    if [[ $failed -eq 0 ]]; then
        rm -f "$plan_file"
        _sqn_success "Upgrade complete. $upgraded tools installed/updated."
    else
        _sqn_warn "Upgrade completed with $failed errors. $upgraded tools installed."
    fi

    _sqn_refresh_manifest
}

# ============================================================
# BIN CLEANER — move non-binary clutter out of arch bin dir
# ============================================================
_sqn_clean_bin() {
    local bin="$SQN_ARCH_BIN"
    [[ -d "$bin" ]] || return 0

    local themes_dir="$bin/themes"
    local man_dir="$bin/man"
    local completions_dir="$bin/completions"
    local docs_dir="$bin/docs"

    mkdir -p "$themes_dir" "$man_dir" "$completions_dir" "$docs_dir"

    find "$bin" -maxdepth 1 -name "*.theme" | while read -r f; do
        mv "$f" "$themes_dir/" 2>/dev/null && echo "  Moved: $(basename "$f") → themes/"
    done

    find "$bin" -maxdepth 1 \( -name "*.1" -o -name "*.8" \) 2>/dev/null | while read -r f; do
        mv "$f" "$man_dir/" 2>/dev/null && echo "  Moved: $(basename "$f") → man/"
    done

    for f in "$bin"/zoxide.bash "$bin"/zoxide.elv "$bin"/zoxide.fish \
              "$bin"/_zoxide "$bin"/_zoxide.ps1 "$bin"/autocomplete \
              "$bin"/complete; do
        [[ -e "$f" ]] && mv "$f" "$completions_dir/" 2>/dev/null && echo "  Moved: $(basename "$f") → completions/"
    done

    for f in "$bin"/LICENSE* "$bin"/COPYING "$bin"/UNLICENSE \
              "$bin"/doc "$bin"/icon.png "$bin"/icon.svg; do
        [[ -e "$f" ]] && mv "$f" "$docs_dir/" 2>/dev/null && echo "  Moved: $(basename "$f") → docs/"
    done

    rmdir "$themes_dir" "$man_dir" "$completions_dir" "$docs_dir" 2>/dev/null || true

    _sqn_success "Bin directory cleaned"
}

# ============================================================
# VERIFICATION
# ============================================================
_sqn_verify() {
    _sqn_info "Verifying installed tools..."

    local ok=0 fail=0
    for tool in "${!SQN_UPSTREAM_REPO[@]}"; do
        local path="${SQN_ARCH_BIN}/$tool"
        echo -n "  $tool... "
        if [[ -x "$path" ]]; then
            echo "OK"
            ((ok++))
        else
            echo "MISSING"
            ((fail++))
        fi
    done

    echo ""
    echo "Verified: $ok OK, $fail missing"

    if [[ $fail -gt 0 ]]; then
        _sqn_warn "Run 'sqn update && sqn upgrade' to install missing tools"
        return 1
    fi

    _sqn_success "All tools verified"
}

# SQN Core Helper
# DEPENDS:

sqn_alias_ll() {
    if [[ -x "${SQN_ARCH_BIN}/eza" ]]; then
        alias sqn_ll='${SQN_ARCH_BIN}/eza -la --group-directories-first --git'
    else

_sqn_load_helpers() {
    local load_order=""
    local errors=0

    shopt -s nullglob 2>/dev/null || true

    for helper in "$SQN_HELPERS"/*.sh; do
        [[ -r "$helper" ]] || continue
        local basename_helper=$(basename "$helper")

        if [[ -z "${SQN_HELPER_ALLOWLIST[$basename_helper]:-}" ]]; then
            _sqn_warn "Helper not in allowlist: $basename_helper (skipped)"
            continue
        fi

        if ! bash -n "$helper" 2>/dev/null; then
            _sqn_error "Syntax error in helper: $basename_helper — not loaded"
            ((errors++))
            continue
        fi

        load_order="$load_order $helper"
    done

    for helper in $load_order; do
        source "$helper" 2>/dev/null || {
            _sqn_error "Failed to load: $(basename "$helper")"
            ((errors++))
        }
    done

    shopt -u nullglob 2>/dev/null || true

    [[ $errors -eq 0 ]] || return 1
    return 0
}

_sqn_init_language() {
    mkdir -p "$SQN_LIB"
    if [[ ! -f "$SQN_LIB/language.sh" ]]; then
        cat > "$SQN_LIB/language.sh" << 'LANGEOF'
#!/bin/bash
# SQN Language Layer — Plain English, Snake_case, Real bash

# ---- NAVIGATION ----
go() {
    case "${1:-}" in
        home)    cd ~ ;;
        back)    cd - ;;
        up)      cd .. ;;
        storage) cd /media 2>/dev/null || echo "No /media found" ;;
        *)       [[ -n "${1:-}" ]] && cd "$1" || cd ~ ;;
    esac
}
whereami() { pwd; }

# ---- FILES ----
show() {
    case "${1:-}" in
        files)      _sqn_ls_smart ;;
        everything) _sqn_ls_smart -a ;;
        hidden)     find "${2:-.}" -maxdepth 1 -name ".*" -not -name "." 2>/dev/null ;;
        big)        _sqn_du_smart "${2:-.}" 2>/dev/null | head -20 ;;
        *)          _sqn_ls_smart "${1:-.}" ;;
    esac
}
getfile()  { _sqn_find_normalized "${1:-}" "${2:-.}"; }
gettext()  { _sqn_grep_smart -r "${1:-}" "${2:-.}" 2>/dev/null; }
copy()     { cp -r "${1:?}" "${2:?}" && echo "Copied."; }
move()     { mv "${1:?}" "${2:?}" && echo "Moved."; }
make() {
    case "${1:-}" in
        folder|dir) mkdir -p "${2:?}" && echo "Created." ;;
        file)       touch "${2:?}" && echo "Created." ;;
        *)          command make "$@" ;;
    esac
}
nuke() {
    [[ -z "${1:-}" ]] && echo "Nuke what?" && return 1
    echo -n "Nuke '$1'? [y/N] " && read -r c
    [[ "$c" =~ ^[Yy]$ ]] && rm -rf "$1" && echo "Gone."
}

# ---- SYSTEM ----
whatsrunning() { _sqn_ps_smart; }
whatseating() {
    case "${1:-}" in
        cpu) ps aux --sort=-%cpu | head -15 ;;
        ram) ps aux --sort=-%mem | head -15 ;;
        *)   _sqn_ps_smart | head -15 ;;
    esac
}
howfull() { _sqn_df_smart; }
online()  { ping -c 1 -W 2 8.8.8.8 >/dev/null && echo "✓ Online" || echo "✗ Offline"; }
sysmon()  { btop 2>/dev/null || htop 2>/dev/null || top; }
kill_process() {
    local target="${1:-}"
    [[ -z "$target" ]] && echo "kill_process <PID or name>" && return 1
    if [[ "$target" =~ ^[0-9]+$ ]]; then command kill "$target"
    else pkill -f "$target"
    fi
}

refresh() { source "$SQN_ROOT/sqn" && echo "SQN Shell reloaded."; }

    case "${1:-help}" in

            echo ""
            echo "Engine:"
            echo "  sqn bootstrap    - initialize directories"
            echo "  sqn update       - query upstream, build upgrade plan"
            echo "  sqn upgrade      - download and install tools"
            echo "  sqn verify       - check installed tool integrity"
            echo "  sqn doctor       - full diagnostics"
            echo "  sqn status       - show current state"
            echo "  sqn clean        - clear cache"
            echo "  sqn clean-bin    - clean up bin directory"
            echo "  sqn unload       - restore host shell state"
            echo "  sqn reload       - reload SQN Shell"
            echo "  sqn run <cmd>    - run command in SQN environment"
            echo "  sqn explain <cmd>- show help for a command"
            echo ""
            echo "Language: go, whereami, show, getfile, gettext,"
            echo "          copy, move, make, nuke, whatsrunning,"
            echo "          whatseating, howfull, online, sysmon,"
            echo "          kill_process, refresh"
            echo ""
            echo "Helpers: $(ls "$SQN_HELPERS" 2>/dev/null | tr '\n' ' ')"
            ;;
        *) _sqn_engine_command "$@" ;;
    esac
}

    fi
}

_sqn_explain() {
    local command="${1:-}"

    declare -A explanations=(
        ["go"]="Navigate directories. go home|back|up|<path>"
        ["whereami"]="Show current working directory"
        ["show"]="List files. show [files|everything|hidden|big|path]"
        ["getfile"]="Find files by name. getfile <pattern> [path]"
        ["gettext"]="Search text in files. gettext <pattern> [path]"
        ["copy"]="Copy files/directories. copy <src> <dst>"
        ["move"]="Move files/directories. move <src> <dst>"
        ["make"]="Create things. make [folder|file] <name>"
        ["nuke"]="Delete with confirmation. nuke <path>"
        ["whatsrunning"]="Show running processes"
        ["whatseating"]="Show resource usage. whatseating [cpu|ram]"
        ["howfull"]="Show disk usage"
        ["online"]="Check internet connectivity"
        ["sysmon"]="Launch system monitor"
        ["kill_process"]="Terminate process. kill_process <PID|name>"
        ["refresh"]="Reload SQN environment"
        ["bootstrap"]="Initialize SQN directories and environment"
        ["update"]="Query upstream for latest tool versions"
        ["upgrade"]="Download and install latest tools"
        ["verify"]="Check installed tool integrity"
        ["doctor"]="Run full system diagnostics"
        ["status"]="Show current SQN state"
        ["clean"]="Clear cache"
        ["clean-bin"]="Clean up extracted bin clutter"
        ["unload"]="Restore host shell state"
        ["reload"]="Reload SQN configuration"
        ["run"]="Execute command in SQN environment"
        ["explain"]="Show help for a command"
        ["version"]="Show version information"
    )

    if [[ -z "$command" ]]; then
        echo -e "\033[1;36mSQN Explain — Available Commands\033[0m"
        echo ""
        for cmd in "${!explanations[@]}"; do
            printf "  %-12s %s\n" "$cmd" "${explanations[$cmd]}"
        done
        echo ""
        echo "Usage: sqn explain <command>"
        return 0
    fi

    if [[ -n "${explanations[$command]:-}" ]]; then
        echo -e "\033[1;36m$command\033[0m — ${explanations[$command]}"
    else
        _sqn_error "No explanation found for: $command"
        echo "Try: sqn explain (to list all commands)"
    fi
}

_sqn_prompt() {
    local code=$?
    local git_branch=""
    
    # One Git call that serves as both test and assignment
    # Works for regular repos, submodules, worktrees, and bare repos
    git_branch=$(git rev-parse --abbrev-ref HEAD 2>/dev/null)

    PS1="\n\e[1;35m[SQN]\e[1;37m[\e[1;32m\w\e[33m${git_branch:+ ($git_branch)}\e[1;37m] \$?=$code\n\e[1;36m→ \e[0m"
}

    if [[ -n "${PROMPT_COMMAND:-}" ]] && [[ "$PROMPT_COMMAND" != *"_sqn_prompt"* ]]; then
        PROMPT_COMMAND="${PROMPT_COMMAND}; _sqn_prompt"
    else
        PROMPT_COMMAND='_sqn_prompt'
    fi
}

# ============================================================
# UNLOAD
# ============================================================
_sqn_unload() {
    if [[ -n "${SQN_OLD_PATH:-}" ]]; then
        export PATH="$SQN_OLD_PATH"
    fi

    if [[ -n "${SQN_OLD_PROMPT_COMMAND:-}" ]]; then
        export PROMPT_COMMAND="$SQN_OLD_PROMPT_COMMAND"
    else
        unset PROMPT_COMMAND
    fi

    echo -e "[1;32m✓ ${SQN_NAME} unloaded. Host shell restored.[0m"
    unset -v SQN_ROOT SQN_BIN SQN_LIB SQN_HELPERS SQN_CACHE SQN_SESSIONS SQN_LOG SQN_VERSION
    unset -v SQN_ARCH SQN_ARCH_BIN SQN_DISTRO SQN_DISPLAY SQN_PLATFORM SQN_NAME
    unset -v SQN_PYTHON SQN_PYTHON_BIN SQN_C SQN_C_BIN SQN_RUST SQN_RUST_BIN
    unset -v SQN_CARGO_BIN SQN_NODE SQN_NODE_BIN SQN_OLD_PATH SQN_OLD_PROMPT_COMMAND
    unset -v SQN_ROOT_ACCESS SQN_ALLOW_SUDO SQN_DRY_RUN
}

# ============================================================
# ENGINE COMMAND
# ============================================================
_sqn_engine_command() {
    case "${1:-}" in
        bootstrap)
            _sqn_bootstrap
            ;;
        update)
            _sqn_plan_update
            ;;
        upgrade)
            if [[ ! -f "$SQN_CACHE/upgrade_plan.tsv" ]]; then
                _sqn_error "No upgrade plan found. Run 'sqn update' first."
                return 1
            fi
            _sqn_apply_upgrade
            ;;
        verify)
            _sqn_verify
            ;;
        status)
            _sqn_status
            ;;
        doctor)
            _sqn_doctor
            ;;
        clean)
            _sqn_info "Cleaning cache..."
            rm -rf "$SQN_CACHE"/*
            _sqn_success "Cache cleared"
            ;;
        clean-bin)
            _sqn_clean_bin
            ;;
        unload)
            _sqn_unload
            ;;
        reload)
            _sqn_info "Reloading ${SQN_NAME}..."
            source "$SQN_ROOT/sqn"
            _sqn_success "Reloaded"
            ;;
        run)
            shift
            if [[ $# -eq 0 ]]; then
                _sqn_error "Usage: sqn run <command> [args...]"
                return 1
            fi
            PATH="$SQN_ARCH_BIN:$SQN_BIN:$PATH" "$@"
            ;;
        explain)
            shift
            _sqn_explain "$@"
            ;;
        version|--version|-V)
            echo "${SQN_NAME} v${SQN_VERSION}"
            echo "Platform: ${SQN_PLATFORM}"
            echo "Arch: ${SQN_ARCH}"
            echo "Distro: ${SQN_DISTRO}"
            ;;
        help|--help|-h)
            _sqn_explain
            ;;
        *)
            _sqn_error "Unknown command: $1. Try 'sqn help' or 'sqn explain'"
            return 1
            ;;
    esac
}
